const audioPath = ""; const sounds = { intro: new Audio(audioPath + "naamloos.wav"), start: new Audio(audioPath + "START.mp3"), stop0: new Audio(audioPath + "stop 0.555 sec.mp3"), stop3: new Audio(audioPath + "STOP 3.mp3"), stop4: new Audio(audioPath + "STOP 4.625 SEC.mp3"), finish: new Audio(audioPath + "aftellen van 1 naar 0.mp3") }; for(let i=2; i<=10; i++) { let fileName = (i === 3) ? "aftellen van 3 naar 1.mp3" : `aftellen ${i} naar 1.mp3`; sounds[`count${i}`] = new Audio(audioPath + fileName); } let queue = []; let currentTimer; let isPlayingCountdown = false; function buildPhases(introDur) { queue.push({action: "WACHT", color: "red", dur: introDur, next: 15}); for (let i = 15; i >= 3; i--) { queue.push({action: "START", color: "green", dur: i, next: i}); queue.push({action: "STOP", color: "red", dur: i, next: (i > 3 ? i-1 : 3)}); } for (let i = 3; i <= 15; i++) { queue.push({action: "START", color: "green", dur: i, next: i}); queue.push({action: "STOP", color: "red", dur: i, next: (i < 15 ? i+1 : 10)}); } } function init() { document.getElementById('ui').innerHTML = `
LADEN...
--
`; sounds.intro.oncanplaythrough = () => { buildPhases(Math.floor(sounds.intro.duration) || 180); startCycle(); }; sounds.intro.load(); } function startCycle() { if (currentTimer) clearInterval(currentTimer); if (queue.length === 0) { location.reload(); return; } const step = queue.shift(); let time = step.dur; updateUI(step); // STOP ALLE AUDIO bij nieuwe fase om chaos te voorkomen Object.values(sounds).forEach(s => { s.pause(); s.currentTime = 0; }); if(step.action === "WACHT") { sounds.intro.play(); } currentTimer = setInterval(() => { document.getElementById('timer').innerText = time; // Countdown logica: NIET bij WACHT, alleen tussen 2 en 10 seconden if (step.action !== "WACHT" && time <= 10 && time > 1) { if (sounds[`count${time}`]) sounds[`count${time}`].play(); } if(time === 1) { sounds.finish.play(); if(queue[0]) { if(queue[0].action === "START") setTimeout(() => sounds.start.play(), 800); if(queue[0].action === "STOP") { let d = queue[0].dur; let s = (d < 4) ? sounds.stop0 : (d < 6 ? sounds.stop3 : sounds.stop4); setTimeout(() => s.play(), 800); } } } if(time <= 0) { clearInterval(currentTimer); startCycle(); } time--; }, 1000); } function updateUI(step) { document.body.style.backgroundColor = step.color; document.body.style.color = (step.color === 'red') ? 'yellow' : 'black'; document.getElementById('action').innerText = step.action; }
